CREATE TABLE STTM_DORI_MASTER_FCUST(
RELATIONSHIP VARCHAR2(35),
DESCRIPTION VARCHAR2(35),
CATEGORY VARCHAR2(35),
PRODUCT_RESTRICTION VARCHAR2(35),
maker_id            VARCHAR2(12),
checker_id          VARCHAR2(12),
checker_dt_stamp    DATE,
maker_dt_stamp      DATE,
mod_no              NUMBER(4),
record_stat         CHAR(1) DEFAULT 'O',
auth_stat           CHAR(1) DEFAULT 'U',
once_auth           CHAR(1) DEFAULT 'N',
CONSTRAINT PK_DORI_MASTER_FCUST PRIMARY KEY (RELATIONSHIP,CATEGORY)
);

CREATE OR REPLACE SYNONYM STTMS_DORI_MASTER_FCUST FOR STTM_DORI_MASTER_FCUST;
